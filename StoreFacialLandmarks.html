def capture_train_face_landmarks_FR(FraudCust,FraudCustName,option,W):
    
    global known_face_encode_error_FR
    global known_face_landmarks_FR
        
    Upsample      = 1
    FraudCust     = imutils.resize(FraudCust, width=W)
    gray          = cv2.cvtColor(FraudCust, cv2.COLOR_BGR2GRAY) 
    clahe         = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(5,8))
    gray          = clahe.apply(gray)
    gray          = cv2.bilateralFilter(gray,5,75,75)
    
    if option == 0:
        rects         = hogFaceDetector(gray, Upsample)  
    if option == 2:
        rects         = cnnFaceDetector(gray, Upsample)

    #################################################################################################################
    # For detected face, find the bounding boxes and landmarks, if un-detected, append into error array to re-process
    #################################################################################################################
    if (len(rects) > 0):
        
        for rect in rects:
            #faceAligned    = face_aligner.align(FraudCust, gray, rect)

            if option == 0:
                face_locations = fr.face_locations(gray,model="hog")   
                
            if option == 2:
                face_locations = fr.face_locations(gray,model="cnn")

            try:
                known_face_landmarks_FR[FraudCustName] = fr.face_encodings(FraudCust,face_locations)[0]  
            except Exception as e:
                known_face_encode_error_FR.append(FraudCustName)
            break
    else:
        known_face_encode_error_FR.append(FraudCustName)
                
    return()

def cropFace():

    global known_face_crop_error

    known_face_landmarks        = {}
    known_face_crop_error       = []
    processed                   = 0

    starttime                   = time.time()

    ##########################
    # Clear the preview folder
    ##########################

    for i in os.listdir(path+'preview/'):
        os.remove(path+'preview/' + i)

    for dirpath, dnames, FraudCusts in os.walk(path +"train/"):

        for FraudCust in FraudCusts:

            try:
                crop_face     = cropface(path +"train/"+FraudCust)   

                if crop_face is None:
                    known_face_crop_error.append(FraudCust)
                else:
                    crop_face =  cv2.cvtColor(crop_face,cv2.COLOR_BGR2RGB)
                    processed +=1
                    cv2.imwrite(path + 'preview/' + FraudCust + '.jpg',crop_face)

            except Exception as e:
                    known_face_crop_error.append(FraudCust)


    print("Processed {0}% (i.e) {1} known faces completed face cropping. Error in capturing {2} known faces".format(str(round(processed/(processed + len(known_face_crop_error)) * 100)),processed,len(known_face_crop_error)))
    print('Time taken = {} Min'.format(round((time.time() - starttime)/60)))
